Convert Time to Double

 for i = 1:length(Time)

     t(i,1) =  str2num(Time{i}(1:2))*60*60+str2num(Time{i}(4:5))*60+str2num(Time{i}(7:8));

     t(i,1) = t(i,1) + str2num(Time{i}(10:13))*1e-4;

 end

 start = t(1);

 for i = 1:length(t)

     t(i) = t(i) - start;

 end
